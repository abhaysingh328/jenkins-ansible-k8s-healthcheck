pipeline {
    agent any

    environment {
        DOCKER_IMAGE = "asabhaysingh328/ansible-node"
        NAMESPACE = "ansible-lab"
        INVENTORY = "hosts.ini"
    }

    stages {

        stage('Checkout Repo') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/abhaysingh328/jenkins-ansible-k8s-healthcheck.git'
            }
        }

        // ✅ FIX: Always build & push (no skip logic)
        stage('Build & Push Docker Image') {
            steps {
                sh '''
                echo "Building Docker image..."
                docker build -t $DOCKER_IMAGE:latest .

                echo "Pushing Docker image..."
                docker push $DOCKER_IMAGE:latest
                '''
            }
        }

        stage('Verify Namespace') {
            steps {
                sh '''
                kubectl get ns $NAMESPACE || kubectl create namespace $NAMESPACE
                '''
            }
        }

        stage('Deploy Pods') {
            steps {
                sh '''
                kubectl apply -f ansible-nodes.yaml -n $NAMESPACE
                '''
            }
        }

        stage('Wait for Pods Ready') {
            steps {
                sh '''
                kubectl wait --for=condition=Ready pod --all -n $NAMESPACE --timeout=180s
                sleep 20
                '''
            }
        }

        // ✅ Debug step (very useful)
        stage('Debug SSH') {
            steps {
                sh '''
                POD=$(kubectl get pods -n $NAMESPACE -o jsonpath="{.items[0].metadata.name}")
                
                echo "Checking sshd process..."
                kubectl exec -n $NAMESPACE $POD -- ps aux | grep sshd || true

                echo "Checking port 22..."
                kubectl exec -n $NAMESPACE $POD -- ss -tulnp | grep 22 || true
                '''
            }
        }

        stage('Run Ansible Health Check') {
            steps {
                sh '''
                export ANSIBLE_HOST_KEY_CHECKING=False
                ansible-playbook -i $INVENTORY healthcheck.yml
                '''
            }
        }

        stage('Install Nginx via Ansible') {
            steps {
                sh '''
                ansible-playbook -i $INVENTORY install-nginx.yml
                '''
            }
        }

        stage('Wait After Nginx Installation') {
            steps {
                sh 'sleep 30'
            }
        }

        stage('Verify Nginx Health') {
            steps {
                sh '''
                ansible -i $INVENTORY servers -m command -a "curl -I http://localhost"
                '''
            }
        }
    }
}
