pipeline {
    agent any
    stages {
        stage('Checkout Repo') {
            steps {
                // Clone your GitHub repo
                git branch: 'main',
                    url: 'https://github.com/abhaysingh328/jenkins-ansible-k8s-healthcheck.git'
            }
        }
        stage('Deploy Pods') {
            steps {
                sh 'kubectl apply -f ansible-nodes.yaml'
            }
        }
        stage('Run Ansible Health Check') {
            steps {
                sh 'ansible-playbook -i hosts.ini healthcheck.yml'
            }
        }
        stage('Install Nginx via Ansible') {
            steps {
                sh 'ansible-playbook -i hosts.ini install-nginx.yml'
            }
        }
    }
}
